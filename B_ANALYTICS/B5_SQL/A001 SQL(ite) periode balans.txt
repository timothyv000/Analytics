Hieronder SQL(ite) statement tbv periodebalans. Per periode wordt middels een SUM i.c.m. CASE een bedrag (amnt) toebehorend aan een periode opgeteld indien het periode nummer overeenkomt.
Hier wordt uitgegaan van 12 perioden. 

SELECT Transactieregels.accID, Grootboek.accDesc, Grootboek.accTp, SUM (CASE WHEN Transactieregels.periodNumber = '0' THEN CASE WHEN Transactieregels.amntTp='D' THEN Transactieregels.amnt ELSE Transactieregels.amnt * -1 END ELSE 0 END) AS "P0", SUM (CASE WHEN Transactieregels.periodNumber = '1' THEN CASE WHEN Transactieregels.amntTp='D' THEN Transactieregels.amnt ELSE Transactieregels.amnt * -1 END ELSE 0 END) AS "P1", SUM (CASE WHEN Transactieregels.periodNumber = '2' THEN CASE WHEN Transactieregels.amntTp='D' THEN Transactieregels.amnt ELSE Transactieregels.amnt * -1 END ELSE 0 END) AS "P2", SUM (CASE WHEN Transactieregels.periodNumber = '3' THEN CASE WHEN Transactieregels.amntTp='D' THEN Transactieregels.amnt ELSE Transactieregels.amnt * -1 END ELSE 0 END) AS "P3", SUM (CASE WHEN Transactieregels.periodNumber = '4' THEN CASE WHEN Transactieregels.amntTp='D' THEN Transactieregels.amnt ELSE Transactieregels.amnt * -1 END ELSE 0 END) AS "P4", SUM (CASE WHEN Transactieregels.periodNumber = '5' THEN CASE WHEN Transactieregels.amntTp='D' THEN Transactieregels.amnt ELSE Transactieregels.amnt * -1 END ELSE 0 END) AS "P5", SUM (CASE WHEN Transactieregels.periodNumber = '6' THEN CASE WHEN Transactieregels.amntTp='D' THEN Transactieregels.amnt ELSE Transactieregels.amnt * -1 END ELSE 0 END) AS "P6", SUM (CASE WHEN Transactieregels.periodNumber = '7' THEN CASE WHEN Transactieregels.amntTp='D' THEN Transactieregels.amnt ELSE Transactieregels.amnt * -1 END ELSE 0 END) AS "P7", SUM (CASE WHEN Transactieregels.periodNumber = '8' THEN CASE WHEN Transactieregels.amntTp='D' THEN Transactieregels.amnt ELSE Transactieregels.amnt * -1 END ELSE 0 END) AS "P8", SUM (CASE WHEN Transactieregels.periodNumber = '9' THEN CASE WHEN Transactieregels.amntTp='D' THEN Transactieregels.amnt ELSE Transactieregels.amnt * -1 END ELSE 0 END) AS "P9", SUM (CASE WHEN Transactieregels.periodNumber = '10' THEN CASE WHEN Transactieregels.amntTp='D' THEN Transactieregels.amnt ELSE Transactieregels.amnt * -1 END ELSE 0 END) AS "P10", SUM (CASE WHEN Transactieregels.periodNumber = '11' THEN CASE WHEN Transactieregels.amntTp='D' THEN Transactieregels.amnt ELSE Transactieregels.amnt * -1 END ELSE 0 END) AS "P11", SUM (CASE WHEN Transactieregels.periodNumber = '12' THEN CASE WHEN Transactieregels.amntTp='D' THEN Transactieregels.amnt ELSE Transactieregels.amnt * -1 END ELSE 0 END) AS "P12", SUM (CASE WHEN Transactieregels.amntTp='D' THEN Transactieregels.amnt ELSE Transactieregels.amnt * -1 END) AS Saldo FROM Transactieregels INNER JOIN Grootboek ON (Transactieregels.accID=Grootboek.accID) GROUP BY Transactieregels.accID
